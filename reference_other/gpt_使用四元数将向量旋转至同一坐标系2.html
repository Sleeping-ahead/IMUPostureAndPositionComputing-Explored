已知两个不同的向量x1、x2和x3，以及表示它们位姿的四元数q1、q2和q3，如何将向量x2和x3旋转变换到x1的坐标系下？请在不引用Eigen库的情况下，用C++写一个示例

在不引用Eigen库的情况下，可以手动实现四元数的乘法和向量的旋转变换。 首先，需要定义一个四元数的结构体，包含四个分量w、x、y、z：

plaintext
struct Quaternion {
    double w, x, y, z;
};

接下来，实现四元数的乘法运算：

plaintext
Quaternion multiply(const Quaternion& q1, const Quaternion& q2) {
    Quaternion result;
    result.w = q1.w*q2.w - q1.x*q2.x - q1.y*q2.y - q1.z*q2.z;
    result.x = q1.w*q2.x + q1.x*q2.w + q1.y*q2.z - q1.z*q2.y;
    result.y = q1.w*q2.y - q1.x*q2.z + q1.y*q2.w + q1.z*q2.x;
    result.z = q1.w*q2.z + q1.x*q2.y - q1.y*q2.x + q1.z*q2.w;
    return result;
}

然后，实现向量的旋转变换。首先将向量转换为四元数形式，即将向量的三个分量作为虚部，第四个分量为0作为实部。然后将向量对应的四元数和位姿对应的四元数相乘，再将结果四元数和位姿的共轭四元数相乘，最后将结果四元数的虚部作为旋转变换后的向量。

plaintext
Quaternion rotateVector(const Quaternion& q, const double& x, const double& y, const double& z) {
    Quaternion v;
    v.w = 0;
    v.x = x;
    v.y = y;
    v.z = z;
    Quaternion result = multiply(q, multiply(v, Quaternion{q.w, -q.x, -q.y, -q.z}));
    return Quaternion{0, result.x, result.y, result.z};
}

最后，在main函数中调用rotateVector函数，将x2和x3分别旋转变换到x1的坐标系下：

plaintext
int main() {
    Quaternion q1 = {1, 0, 0, 0}; // q1为单位四元数，表示不旋转
    Quaternion q2 = {...}; // q2为表示x2位姿的四元数
    Quaternion q3 = {...}; // q3为表示x3位姿的四元数
    double x1 = ...;
    double y1 = ...;
    double z1 = ...;
    double x2 = ...;
    double y2 = ...;
    double z2 = ...;
    double x3 = ...;
    double y3 = ...;
    double z3 = ...;
    Quaternion q1_conj = {q1.w, -q1.x, -q1.y, -q1.z}; // q1的共轭四元数
    Quaternion q2_to_1 = multiply(q1, multiply(q2, q1_conj)); // 将x2从q2的坐标系变换到q1的坐标系
    Quaternion q3_to_1 = multiply(q1, multiply(q3, q1_conj)); // 将x3从q3的坐标系变换到q1的坐标系
    double x2_new = rotateVector(q2_to_1, x2, y2, z2).x; // 旋转变换后的x2
    double y2_new = rotateVector(q2_to_1, x2, y2, z2).y;
    double z2_new = rotateVector(q2_to_1, x2, y2, z2).z;
    double x3_new = rotateVector(q3_to_1, x3, y3, z3).x; // 旋转变换后的x3
    double y3_new = rotateVector(q3_to_1, x3, y3, z3).y;
    double z3_new = rotateVector(q3_to_1, x3, y3, z3).z;
    return 0;
}
